Discussion of Activity 5.7 - Using a LinkedList

Here is our sample solution for the class LinkedStack.

import java.util.*;

public class LinkedStack implements Stack
{
   private LinkedList<Object> theList;

   public LinkedStack()
   {
      theList = new LinkedList<Object>();
   }

   public void push(Object o)
   {
      theList.addLast(o);
   }

   public Object pop()
   {
      return theList.removeLast();
   }

   public Object peek()
   {
      return theList.getLast();
   }

   public boolean isEmpty()
   {
      return theList.isEmpty();
   }
}
